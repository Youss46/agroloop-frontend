## 5. Create src/hooks/useSEO.js

```js
import { useEffect } from 'react'

export const useSEO = ({ title, description, url }) => {
  useEffect(() => {
    // Title
    document.title = title 
      ? `${title} | AgroLoopCI` 
      : 'AgroLoopCI — Zéro gaspillage, Impact maximum'
    
    // Description
    const desc = document.querySelector('meta[name="description"]')
    if (desc && description) desc.setAttribute('content', description)

    // OG tags
    const ogTitle = document.querySelector('meta[property="og:title"]')
    const ogDesc = document.querySelector('meta[property="og:description"]')
    const ogUrl = document.querySelector('meta[property="og:url"]')
    if (ogTitle && title) ogTitle.setAttribute('content', `${title} | AgroLoopCI`)
    if (ogDesc && description) ogDesc.setAttribute('content', description)
    if (ogUrl && url) ogUrl.setAttribute('content', `https://agroloopci.ci${url}`)
  }, [title, description, url])
}
```
